<theme-toggle>

A custom element that allows you to toggle between dark and light theme.

CSS

:root {
  --button-base: #d0cfcf;
  --button-hover: #9b9b9b;
}

:root[data-theme="dark"] {
  --button-base: #0c151c;
  --button-hover: #040a0f;
}

theme-toggle:not(:defined) {
  display: none;
}

theme-toggle.custom-styles {
  --icon-light-color: #1d1d1d;
  --text-light-color: #1d1d1d;
  --icon-dark-color: #efdb43;
  --text-dark-color: #efdb43;
}

theme-toggle.custom-styles::part(theme-toggle) {
  gap: 1rem;
  border: 0;
  background-color: var(--button-base);
  border-radius: 0.5rem;
  padding: 1rem 2rem;
  line-height: 1;
  cursor: pointer;
}

theme-toggle.custom-styles::part(theme-toggle):hover {
  background-color: var(--button-hover);
}

theme-toggle.custom-styles::part(theme-toggle__icon) {
  width: 30px;
  height: 30px;
}

theme-toggle.custom-styles::part(theme-toggle__text) {
  font-size: 1.8rem;
}
